function readFileAsBase64(file){

  return new Promise((resolve,reject)=>{

    if(file.size>MAX_FILE_SIZE){reject(new Error(`File "${file.name}" vượt quá 3MB`));return}

    const r=new FileReader();

    r.onload=e=>resolve({name:file.name,type:file.type,size:file.size,data:e.target.result,addedAt:today()});

    r.onerror=()=>reject(new Error('Lỗi đọc file'));

    r.readAsDataURL(file);

  });

}

    setLoading(true);

    try{

      const added=await Promise.all(Array.from(fileList).map(readFileAsBase64));

function SchemaBuilder({db,refresh,toast}){

  const [kind,setKind]=useState('report');

  const [sections,setSections]=useState(JSON.parse(JSON.stringify(db.reportSections)));

  useEffect(()=>setSections(JSON.parse(JSON.stringify(kind==='report'?db.reportSections:db.adhocSections))),[kind,db]);

 

  function addField(si){

    const id=prompt('Mã trường (không dấu, không cách, ví dụ: new_field):');

    if(!id||!/^[a-z_0-9]+$/.test(id)){toast('error','Mã trường chỉ gồm chữ thường, số và dấu _');return}

    const label=prompt('Tên hiển thị của trường:')||id;

    const newSec=[...sections];newSec[si].fields.push({id,label,type:'textarea'});setSections(newSec);

  }

  function removeField(si,fi){const s=[...sections];s[si].fields.splice(fi,1);setSections(s);}

  function saveSchema(){

    if(kind==='report')DB.reportSections=sections;else DB.adhocSections=sections;

    save();toast('success','Đã lưu cấu hình lên Cloud');

  }

 

  return(

    <main style={{padding:28,maxWidth:1050,margin:'0 auto',width:'100%'}}>

      <h2 style={{fontSize:21,fontWeight:1000}}>🧩 Cấu hình trường nội dung</h2>

      <div style={{display:'flex',gap:9,marginBottom:18}}>

        <Btn kind={kind==='report'?'primary':'soft'} onClick={()=>setKind('report')}>📝 Báo cáo định kỳ</Btn>

        <Btn kind={kind==='adhoc'?'primary':'soft'} onClick={()=>setKind('adhoc')}>🚨 Báo cáo đột xuất</Btn>

        <Btn kind="ok" onClick={saveSchema}>💾 Lưu cấu hình Cloud</Btn>

      </div>

      {sections.map((s,si)=>(

        <Card key={s.id} style={{padding:20,marginBottom:16}}>

          <h3 style={{fontWeight:1000,fontSize:15,marginBottom:13}}>{s.title}</h3>

          <div style={{display:'flex',flexDirection:'column',gap:9}}>

            {s.fields.map((f,fi)=>(

              <div key={f.id} style={{display:'grid',gridTemplateColumns:'1fr 1.5fr 120px 100px 36px',gap:9,alignItems:'center'}}>

                <input value={f.id} onChange={e=>{const ns=[...sections];ns[si].fields[fi].id=e.target.value;setSections(ns)}} placeholder="field_id" style={{padding:'8px 10px',border:'1px solid #cbd5e1',borderRadius:8,fontFamily:'monospace',fontSize:12}}/>

                <input value={f.label} onChange={e=>{const ns=[...sections];ns[si].fields[fi].label=e.target.value;setSections(ns)}} placeholder="Tên trường" style={{padding:'8px 10px',border:'1px solid #cbd5e1',borderRadius:8,fontSize:12}}/>

                <select value={f.type} onChange={e=>{const ns=[...sections];ns[si].fields[fi].type=e.target.value;setSections(ns)}} style={{padding:'8px 10px',border:'1px solid #cbd5e1',borderRadius:8,fontSize:12}}>

                  {['number','text','textarea','select'].map(t=><option key={t} value={t}>{t}</option>)}

                </select>

                <label style={{fontWeight:850,fontSize:12,display:'flex',gap:6,cursor:'pointer'}}>

                  <input type="checkbox" checked={!!f.required} onChange={e=>{const ns=[...sections];ns[si].fields[fi].required=e.target.checked;setSections(ns)}}/> Bắt buộc

                </label>

                <button onClick={()=>removeField(si,fi)} style={{background:'#fee2e2',border:'none',borderRadius:7,padding:'7px',color:'#dc2626',fontWeight:900,cursor:'pointer'}}>✕</button>

              </div>

            ))}

    </main>

  );
