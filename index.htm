/* ═══════════════════ FILE UPLOADER UTILS ═══════════════════ */

function FileUploader({files=[],onChange}){

  const ref=useRef();

  const handleFiles=async (fileList)=>{

    const added = await Promise.all(Array.from(fileList).map(file => {

      return new Promise((resolve) => {

        const r = new FileReader();

        r.onload = e => resolve({name:file.name, type:file.type, size:file.size, data:e.target.result, addedAt:today()});

        r.readAsDataURL(file);

      });

    }));

    onChange([...files,...added]);

  };

  return (

    <div style={{marginTop:10}}>

      <div onClick={()=>ref.current.click()} style={{border:'2px dashed #cbd5e1',padding:14,borderRadius:10,textAlign:'center',cursor:'pointer',background:'#f8fafc'}}>

        <input ref={ref} type="file" multiple style={{display:'none'}} onChange={e=>handleFiles(e.target.files)}/>

        <span style={{fontSize:13,color:'#64748b',fontWeight:700}}>📎 Click để đính kèm văn bản / ảnh minh chứng (Mỗi file ≤ 3MB)</span>

      </div>

      <div style={{marginTop:8,display:'flex',flexDirection:'column',gap:6}}>

        {files.map((f,i)=>(

          <div key={i} style={{display:'flex',justifyContent:'space-between',background:'#fff',padding:'6px 12px',borderRadius:8,border:'1px solid #e2e8f0',fontSize:12,fontWeight:700}}>

            <span>{fileIcon(f.type)} {f.name} ({fmtBytes(f.size)})</span>

            <button onClick={()=>onChange(files.filter((_,idx)=>idx!==i))} style={{color:'#dc2626',border:'none',background:'none'}}>✕</button>

          </div>

        ))}

      </div>

    </div>

  );

}

/* ═══════════════════ PHÂN HỆ KHAI BÁO CHI ĐOÀN ═══════════════════ */

function PeriodicView({db, user, selMonth, setSelMonth}){

  const reportKey = `${user.id}_${selMonth}`;

  const report = db.reports?.[reportKey] || {status:'draft', data:{}, files:[]};

  const [formData, setFormData] = useState(report.data || {});

  const [files, setFiles] = useState(report.files || []);

  const [statusMsg, setStatusMsg] = useState('');

 

  useEffect(() => {

    const r = db.reports?.[reportKey] || {data:{}, files:[]};

    setFormData(r.data || {});

    setFiles(r.files || []);

  }, [selMonth, db.reports, reportKey]);

 

  const handleSaveCloud = (isSubmit) => {

    // Đẩy dữ liệu trực tiếp lên nhánh 'reports/reportKey' trên Firebase Cloud

    const reportRef = ref(database, `cong_bao_cao_doan/reports/${reportKey}`);

    const updatedReport = {

      id: reportKey,

      unitId: user.id,

      unitName: user.name,

      month: selMonth,

      status: isSubmit ? 'submitted' : 'draft',

      submittedAt: isSubmit ? today() : '',

      data: formData,

      files: files

    };

 

    set(reportRef, updatedReport).then(() => {

      setStatusMsg(isSubmit ? '🚀 Đã nộp báo cáo trực tuyến thành công!' : '💾 Đã lưu bản nháp lên Cloud!');

      setTimeout(()=>setStatusMsg(''), 3000);

    });

  };

 

  const isSubmitted = report.status === 'submitted';

 

  return (

    <div>

      <div style={{display:'flex',justifyContent:'space-between',marginBottom:16}}>

        <div>

          <h2 style={{fontWeight:900}}>Khai trình báo cáo định kỳ trực tuyến</h2>

          <select style={{padding:6,borderRadius:8,marginTop:6,fontWeight:800}} value={selMonth} onChange={e=>setSelMonth(e.target.value)}>

            {MONTHS.map(m=><option key={m} value={m}>{fmtMonth(m)}</option>)}

          </select>

        </div>

        <div style={{display:'flex',gap:8,alignItems:'center'}}>

          {!isSubmitted && <Btn kind="soft" onClick={()=>handleSaveCloud(false)}>Lưu nháp trên mây</Btn>}

          {!isSubmitted && <Btn onClick={()=>handleSaveCloud(true)}>Nộp báo cáo</Btn>}

          {isSubmitted && <Badge bg="#dcfce7" color="#166534">✓ Đã nộp tập trung</Badge>}

        </div>

      </div>

      {statusMsg && <div style={{padding:12,background:'#dcfce7',color:'#166534',borderRadius:10,fontWeight:800,marginBottom:12}}>{statusMsg}</div>}

     

      {DEFAULT_REPORT_SECTIONS.map(sec=>(

        <Card key={sec.id} style={{padding:20,marginBottom:16,borderTop:`4px solid ${sec.color}`}}>

          <h3 style={{color:sec.color,fontWeight:900,marginBottom:12}}>{sec.title}</h3>

          {sec.fields.map(f=>(

            <Field key={f.id} field={f} value={formData[f.id]} onChange={val => !isSubmitted && setFormData(p=>({...p,[f.id]:val}))}/>

          ))}

        </Card>

      ))}

      <Card style={{padding:20}}>

        <h3 style={{fontWeight:900,marginBottom:10}}>Hồ sơ đính kèm minh chứng</h3>

        {isSubmitted ? (

          files.map((f,i)=><div key={i} style={{fontSize:13,fontWeight:700,color:'#4f46e5'}}>📎 {f.name}</div>)

        ) : <FileUploader files={files} onChange={setFiles}/>}

      </Card>

    </div>

  );

}

                  <tr key={u.id} style={{borderBottom:'1px solid #f1f5f9'}}>

                    <td style={{padding:12}}>

                      {r?.status==='submitted' ? <Badge bg="#dcfce7" color="#166534">Đã nộp</Badge> :

                       r?.status==='draft' ? <Badge bg="#eff6ff" color="#1e40af">Đang viết nháp</Badge> :

                       <Badge bg="#fee2e2" color="#991b1b">Chưa khai báo</Badge>}

                    </td>

                    <td style={{padding:12,fontSize:13,color:'#64748b',fontWeight:700}}>{r?.submittedAt || '--/--/----'}</td>

                    <td style={{padding:12}}>

                      {r ? <Btn kind="soft" onClick={()=>setSelectedReport(r)}>Xem số liệu trực tuyến 👁️</Btn> : <span style={{fontSize:12,color:'#94a3b8',fontWeight:700}}>Không có dữ liệu</span>}

                    </td>

                  </tr>

                );

              })}

            </tbody>

          </table>

        </Card>

 

        {/* Khung Xem Chi Tiết Trực Tuyến Khi Bấm Chọn */}

        {selectedReport && (

          <Card style={{padding:20,borderLeft:'5px solid #7c3aed',height:'fit-content'}}>

            <div style={{display:'flex',justifyContent:'space-between',marginBottom:14}}>

              <h3 style={{fontWeight:900,color:'#7c3aed'}}>{selectedReport.unitName}</h3>

              <button onClick={()=>setSelectedReport(null)} style={{border:'none',background:'none',fontSize:16,fontWeight:900}}>✕ Đóng</button>

            </div>

            <div style={{display:'flex',flexDirection:'column',gap:12,fontSize:13}}>

              <div><b>Tổng số đoàn viên:</b> {selectedReport.data?.total_members || 0}</div>

              <div><b>Đảng viên sinh hoạt Đoàn:</b> {selectedReport.data?.party_members || 0}</div>

              <hr style={{border:'none',borderTop:'1px solid #e2e8f0'}}/>

              <div><b>Giáo dục tư tưởng:</b> <p style={{background:'#f8fafc',padding:8,borderRadius:6,marginTop:4}}>{selectedReport.data?.political_education || 'Chưa khai báo'}</p></div>

              <div><b>Kết quả phong trào:</b> <p style={{background:'#f8fafc',padding:8,borderRadius:6,marginTop:4}}>{selectedReport.data?.movement_result || 'Chưa khai báo'}</p></div>

              {selectedReport.files && (

                <div>

                  <b>Tài liệu minh chứng đính kèm:</b>

                  {selectedReport.files.map((f,i)=>(

                    <div key={i} style={{color:'#4f46e5',marginTop:4,cursor:'pointer',fontWeight:700}} onClick={()=>{

                      const a=document.createElement('a');a.href=f.data;a.download=f.name;a.click();

                    }}>💾 Click tải tệp: {f.name}</div>

                  ))}

                </div>

              )}

            </div>

          </Card>

        )}

      </div>

    </div>

  );

}
